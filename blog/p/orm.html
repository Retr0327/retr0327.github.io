    database query User Post model User Post one-to-many ()

    model User {
       uid     String     @id @default(uuid()) @db.Uuid
       name    String
       post    Post[] @relation("postsToUsers")
    

       @@map("users")
    }
    

    model Post {
      userUid   String   @map("user_uid") @db.Uuid
      title String
      content String
    

      user User @relation("postsToUsers", fields: [userUid], references: [uid], onDelete: NoAction, onUpdate: NoAction, map: "fkey_users")
      @@map("posts")
    }

    const users = await prisma.user.findMany(); //  SQL 
    

    for (const user of users) {    //  SELECT * FROM users
      const posts = await prisma.post.findMany({   //  SELECT * FROM posts

          userUid: user.uid, //  SELECT * FROM users WHERE uid = posts.user_uid
